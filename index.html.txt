<script>

// ===== CONFIG =====
let adminPassword="adminlost";

// Prompt admin
if(prompt("Contraseña admin (dejar vacío si no eres admin)")===adminPassword){
document.getElementById("adminPanel").style.display="block";
}

// ===== INICIALIZACIÓN DE DATOS =====
function initStorage(){
    // Lost media base
    if(!localStorage.getItem("lostMedia")){
        let baseLost=[
            {name:"Lol Superman",estado:"perdido"},
            {name:"No Me Pises Pa",estado:"perdido"},
            {name:"The Most Mysterious Song on the Internet",estado:"perdido"},
            {name:"Everyone Knows That",estado:"perdido"},
            {name:"Clockman",estado:"encontrado"},
            {name:"Cry Baby Lane",estado:"encontrado"},
            {name:"Saki Sanobashi",estado:"perdido"},
            {name:"Polybius",estado:"perdido"},
            {name:"Dead Bart",estado:"perdido"},
            {name:"SuicideMouse.avi",estado:"perdido"},
            {name:"Ben Drowned",estado:"perdido"},
            {name:"Candle Cove",estado:"perdido"},
            {name:"Hitogata",estado:"perdido"}
        ];
        localStorage.setItem("lostMedia",JSON.stringify(baseLost));
    }

    if(!localStorage.getItem("comentarios")) localStorage.setItem("comentarios",JSON.stringify([]));
    if(!localStorage.getItem("pedidos")) localStorage.setItem("pedidos",JSON.stringify([]));
    if(!localStorage.getItem("teorias")) localStorage.setItem("teorias",JSON.stringify([]));
}

// ===== VARIABLES =====
initStorage();

let lostMedia=JSON.parse(localStorage.getItem("lostMedia"));
let comentarios=JSON.parse(localStorage.getItem("comentarios"));
let pedidos=JSON.parse(localStorage.getItem("pedidos"));
let teorias=JSON.parse(localStorage.getItem("teorias"));

// ===== FUNCIONES =====
function guardarCasos(){ localStorage.setItem("lostMedia",JSON.stringify(lostMedia)); }
function guardarComentarios(){ localStorage.setItem("comentarios",JSON.stringify(comentarios)); }
function guardarPedidos(){ localStorage.setItem("pedidos",JSON.stringify(pedidos)); }
function guardarTeorias(){ localStorage.setItem("teorias",JSON.stringify(teorias)); }

function contador(){
let resueltos=lostMedia.filter(x=>x.estado==="encontrado").length;
document.getElementById("contador").innerText=resueltos;
}

function mostrarLista(lista){
let ul=document.getElementById("lostList");
ul.innerHTML="";
lista.forEach((l,i)=>{
    let estadoTexto="";
    if(l.estado==="encontrado") estadoTexto="<span class='encontrado'>✔ ENCONTRADO</span>";
    if(l.estado==="parcial") estadoTexto="<span class='parcial'>⚠ PARCIAL</span>";
    let li=document.createElement("li");
    li.innerHTML=l.name+" "+estadoTexto+
        "<br><button onclick='marcar("+i+",\"encontrado\")'>Encontrado</button>"+
        "<button onclick='marcar("+i+",\"parcial\")'>Parcial</button>";
    ul.appendChild(li);
});
contador();
}

function buscar(){
let t=document.getElementById("search").value.toLowerCase();
let resultados=lostMedia.filter(x=>x.name.toLowerCase().includes(t));
mostrarLista(resultados);
}

function marcar(i,estado){
lostMedia[i].estado=estado;
guardarCasos();
mostrarLista(lostMedia);
}

function agregarCaso(){
let nombre=document.getElementById("nuevoCaso").value;
if(!nombre) return;
lostMedia.push({name:nombre,estado:"perdido"});
guardarCasos();
mostrarLista(lostMedia);
document.getElementById("nuevoCaso").value="";
}

mostrarLista(lostMedia);

// ===== COMENTARIOS =====
function mostrarComentarios(){
let ul=document.getElementById("comentarios"); ul.innerHTML="";
comentarios.forEach((c,i)=>{
    let li=document.createElement("li");
    li.innerHTML="<b>"+c.nombre+"</b>: "+c.texto+
    "<br><button onclick='responder("+i+")'>Responder</button>"+
    "<div>"+(c.respuesta||"")+"</div>";
    ul.appendChild(li);
});
}
function comentar(){
let nombre=document.getElementById("comentarioNombre").value;
let texto=document.getElementById("comentarioTexto").value;
if(!nombre || !texto) return;
comentarios.push({nombre,texto,respuesta:""});
guardarComentarios();
mostrarComentarios();
document.getElementById("comentarioNombre").value="";
document.getElementById("comentarioTexto").value="";
}
function responder(i){
let link=prompt("Pon el link del lost media encontrado");
if(!link) return;
comentarios[i].respuesta="🔎 "+link;
guardarComentarios();
mostrarComentarios();
}
mostrarComentarios();

// ===== TEORIAS =====
function mostrarTeorias(){
let ul=document.getElementById("listaTeorias"); ul.innerHTML="";
teorias.forEach(t=>{
    let li=document.createElement("li");
    li.innerHTML="<b>"+t.nombre+"</b>: "+t.texto;
    ul.appendChild(li);
});
}
function enviarTeoria(){
let nombre=document.getElementById("teoriaNombre").value;
let texto=document.getElementById("teoriaTexto").value;
if(!nombre || !texto) return;
teorias.push({nombre,texto});
guardarTeorias();
mostrarTeorias();
document.getElementById("teoriaNombre").value="";
document.getElementById("teoriaTexto").value="";
}
mostrarTeorias();

// ===== PEDIDOS =====
function mostrarPedidos(){
let lista=document.getElementById("listaPedidos"); lista.innerHTML="";
pedidos.forEach(p=>{
    let li=document.createElement("li");
    li.innerHTML="<b>"+p.titulo+"</b><br>"+p.descripcion;
    lista.appendChild(li);
});
}
function enviarPedido(){
let titulo=document.getElementById("pedidoTitulo").value;
let descripcion=document.getElementById("pedidoDescripcion").value;
if(!titulo) return;
pedidos.push({titulo,descripcion});
guardarPedidos();
mostrarPedidos();
document.getElementById("pedidoTitulo").value="";
document.getElementById("pedidoDescripcion").value="";
}
mostrarPedidos();
